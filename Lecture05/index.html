Local Function

    def addtwo(x, y):
        return x + y

Functions are objects


def get_addtwo():
    def addtwo(x, y):
        return x + y
    return addtwo

addtwo = get_addtwo()
print(addtwo(100, 20)) # 120

Closure


def get_addnum(num):
    def addnum(x):
        return x + num
    return addnum

add10 = get_addnum(10)
print(add10(20)) # 30
